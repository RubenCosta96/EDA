341void sortVehiclesByAutonomy(Vehicle **head)
342{
343 Vehicle *aux = *head;
344 if (aux == NULL)
345 return;
346
347 // create a temporary linked list to hold the sorted nodes
348 Vehicle *sortedList = NULL;
349
350 while (aux != NULL)
351 {
352 // remove the first node from the unsorted list
353 Vehicle *current = aux;
354 aux = current->next;
355
356 // find the appropriate position for the node in the sorted list
357 Vehicle **insertionPoint = &sortedList;
358 while (*insertionPoint != NULL && compare((*insertionPoint)->autonomy, current->autonomy) > 0)
359 {
360 insertionPoint = &((*insertionPoint)->next);
361 }
362
363 // insert the node into the sorted list
364 current->next = *insertionPoint;
365 *insertionPoint = current;
366 }
367
368 // set the head of the original list to the head of the sorted list
369 *head = sortedList;
370}
