288float dijkstra(Graph *graph, char *initial, char *final, float *distances, int id)
289{
290 Graph *aux = graph;
291 Graph *aux1 = graph;
292 int visited[SIZE];
293 int route[SIZE];
294 float weight[SIZE];
295
296 for (int i = 0; i < SIZE; i++)
297 {
298 distances[i] = FLT_MAX;
299 visited[i] = 0;
300 route[i] = -1;
301 weight[i] = 0;
302 }
303
304 int initialIndex = getIdByVertex(graph, initial);
305 distances[initialIndex] = 0;
306
307 float totalDistance = 0.0;
308 while (1)
309 {
310 Graph *minVertex = NULL;
311 float minDistance = INFINITY;
312
313 Graph *current = aux;
314 while (current != NULL)
315 {
316 if (!visited[current->id] && distances[current->id] < minDistance)
317 {
318 minVertex = current;
319 minDistance = distances[current->id];
320 }
321 current = current->next;
322 }
323
324 if (minVertex == NULL || strcmp(minVertex->vertex, final) == 0)
325 {
326 break;
327 }
328
329 visited[minVertex->id] = 1;
330
331 Adjacent *adjacent = minVertex->adjacents;
332 while (adjacent != NULL)
333 {
334
335 int adjacentIndex = getIdByGeocode(aux1, adjacent->vertex);
336 if (!visited[adjacentIndex] && distances[minVertex->id] != FLT_MAX &&
337 distances[minVertex->id] + adjacent->weight < distances[adjacentIndex])
338 {
339 distances[adjacentIndex] = distances[minVertex->id] + adjacent->weight;
340 route[adjacentIndex] = minVertex->id;
341 weight[adjacentIndex] = adjacent->weight;
342 }
343 adjacent = adjacent->next;
344 }
345 }
346 aux = aux->next;
347 int finalIndex = getIdByVertex(graph, final);
348 return printRoute(route, weight, finalIndex, id);
